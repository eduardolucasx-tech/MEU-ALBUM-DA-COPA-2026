rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() {
      return request.auth != null;
    }

    function isFamilyMember() {
      return signedIn()
        && resource.data.members[request.auth.uid].uid == request.auth.uid;
    }

    function joinsExistingFamily(familyCode) {
      return signedIn()
        && resource.data.inviteCode == familyCode
        && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['members', 'updatedAt'])
        && request.resource.data.members.diff(resource.data.members).affectedKeys().hasOnly([request.auth.uid])
        && request.resource.data.members[request.auth.uid].uid == request.auth.uid;
    }

    match /meu_album_copa_v1_family_albums/{familyCode} {
      allow list: if false;
      allow create: if signedIn()
        && request.resource.data.inviteCode == familyCode
        && request.resource.data.ownerUid == request.auth.uid
        && request.resource.data.members[request.auth.uid].uid == request.auth.uid;
      allow get: if isFamilyMember();
      allow update: if isFamilyMember() || joinsExistingFamily(familyCode);
      allow delete: if signedIn() && resource.data.ownerUid == request.auth.uid;
    }
  }
}
